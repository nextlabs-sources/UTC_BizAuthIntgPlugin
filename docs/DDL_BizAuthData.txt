/*** Table Authority ***/
CREATE TABLE Authority(
	InternalLicenseNo VARCHAR(100) NOT NULL,
	OfficialLicenseNo VARCHAR(100) NULL,
	SourceSystem VARCHAR(100) NULL,
	LicenseStatus VARCHAR(10) NULL,
	LicenseType VARCHAR(30) NULL,
	Jurisdiction VARCHAR(10) NULL,
	Scope VARCHAR(30) NULL,
	SME VARCHAR(1) NULL,
	Platform VARCHAR(100) NULL,
	Model VARCHAR(100) NULL,
	StartDate datetime NULL,
	ExpiryDate datetime NULL,
	CONSTRAINT PK_Authority PRIMARY KEY CLUSTERED (InternalLicenseNo ASC )
)
GO
 
/*** Table AuthorityItems ***/
CREATE TABLE AuthorityItems(
	InternalLicenseNo VARCHAR(100) NOT NULL,
	Item VARCHAR(100) NOT NULL
)
GO
